linux内核之原子操作

发表于 2017-09-20 | 分类于 linux内核 |
在任何处理器平台下，都会有一些原子性操作，供操作系统使用，我们这里只讲x86下面的。

在单处理器情况下，每条指令的执行都是原子性的，但在多处理器情况下，只有那些单独的读操作或写操作才是原子性的。为了弥补这一缺点，x86提供了附加的lock前缀，使带lock前缀的读修改写指令也能原子性执行。带lock前缀的指令在操作时会锁住总线，使自身的执行即使在多处理器间也是原子性执行的。xchg指令不带lock前缀也是原子性执行，也就是说xchg执行时默认会锁内存总线。原子性操作是线程间同步的基础，linux专门定义了一种只进行原子操作的类型atomic_t，并提供相关的原子读写调用API。本节就来分析这些原子操作在x86下的实现。

  1. typedef struct {
  2. volatile int counter;
  3. } atomic_t;
    原子类型其实是int类型，只是禁止寄存器对其暂存。
  4. #define ATOMIC_INIT(i) { (i) }
    原子类型的初始化。32位x86平台下atomic API在arch/x86/include/asm/atomic_32.h中实现。
  5. static inline int atomic_read(const atomic_t *v)
  6. {
  7. return v->counter;
  8. }
  10. static inline void atomic_set(atomic_t *v, int i)
  11. {
  12. v->counter = i;
  13. }
    单独的读操作或者写操作，在x86下都是原子性的。
  14. static inline void atomic_add(int i, atomic_t *v)
  15. {
  16. asm volatile(LOCK_PREFIX “addl %1,%0”
  17. : “+m” (v->counter)
  18. : “ir” (i));
  19. }
  21. static inline void atomic_sub(int i, atomic_t *v)
  22. {
  23. asm volatile(LOCK_PREFIX “subl %1,%0”
  24. : “+m” (v->counter)
  25. : “ir” (i));
  26. }
    atomic_add和atomic_sub属于读修改写操作，实现时需要加lock前缀。
  27. static inline int atomic_sub_and_test(int i, atomic_t *v)
  28. {
  29. unsigned char c;
  31. asm volatile(LOCK_PREFIX “subl %2,%0; sete %1”
  32. : “+m” (v->counter), “=qm” (c)
  33. : “ir” (i) : “memory”);
  34. return c;
  35. }
    atomic_sub_and_test执行完减操作后检查结果是否为0。
  36. static inline void atomic_inc(atomic_t *v)
  37. {
  38. asm volatile(LOCK_PREFIX “incl %0”
  39. : “+m” (v->counter));
  40. }
  42. static inline void atomic_dec(atomic_t *v)
  43. {
  44. asm volatile(LOCK_PREFIX “decl %0”
  45. : “+m” (v->counter));
  46. }
    atomic_inc和atomic_dec是递增递减操作。
  47. static inline int atomic_dec_and_test(atomic_t *v)
  48. {
  49. unsigned char c;
  51. asm volatile(LOCK_PREFIX “decl %0; sete %1”
  52. : “+m” (v->counter), “=qm” (c)
  53. : : “memory”);
  54. return c != 0;
  55. }
    atomic_dec_and_test在递减后检查结果是否为0。
  56. static inline int atomic_inc_and_test(atomic_t *v)
  57. {
  58. unsigned char c;
  60. asm volatile(LOCK_PREFIX “incl %0; sete %1”
  61. : “+m” (v->counter), “=qm” (c)
  62. : : “memory”);
  63. return c != 0;
  64. }
    atomic_inc_and_test在递增后检查结果是否为0。
  65. static inline int atomic_add_negative(int i, atomic_t *v)
  66. {
  67. unsigned char c;
  69. asm volatile(LOCK_PREFIX “addl %2,%0; sets %1”
  70. : “+m” (v->counter), “=qm” (c)
  71. : “ir” (i) : “memory”);
  72. return c;
  73. }
    atomic_add_negative在加操作后检查结果是否为负数。
  74. static inline int atomic_add_return(int i, atomic_t *v)
  75. {
  76. int __i;
  77. #ifdef CONFIG_M386
  78. unsigned long flags;
  79. if (unlikely(boot_cpu_data.x86 <= 3))
  80. goto no_xadd;
  81. #endif
  82. / Modern 486+ processor /
  83. __i = i;
  84. asm volatile(LOCK_PREFIX “xaddl %0, %1”
  85. : “+r” (i), “+m” (v->counter)
  86. : : “memory”);
  87. return i + __i;
  89. #ifdef CONFIG_M386
  90. no_xadd: / Legacy 386 processor /
  91. local_irq_save(flags);
  92. __i = atomic_read(v);
  93. atomic_set(v, i + __i);
  94. local_irq_restore(flags);
  95. return i + __i;
  96. #endif
  97. }
    atomic_add_return 不仅执行加操作，而且把相加的结果返回。它是通过xadd这一指令实现的。
  98. static inline int atomic_sub_return(int i, atomic_t *v)
  99. {
  100. return atomic_add_return(-i, v);
  101. }
    atomic_sub_return 不仅执行减操作，而且把相减的结果返回。它是通过atomic_add_return实现的。
  102. static inline int atomic_cmpxchg(atomic_t *v, int old, int new)
  103. {
  104. return cmpxchg(&v->counter, old, new);
  105. }
  107. #define cmpxchg(ptr, o, n) \
  108. ((typeof(*(ptr)))__cmpxchg((ptr), (unsigned long)(o), \
  109. (unsigned long)(n), \
  110. sizeof(*(ptr))))
  112. static inline unsigned long __cmpxchg(volatile void *ptr, unsigned long old,
  113. unsigned long new, int size)
  114. {
  115. unsigned long prev;
  116. switch (size) {
  117. case 1:
  118. asm volatile(LOCK_PREFIX “cmpxchgb %b1,%2”
  119. : “=a”(prev)
  120. : “q”(new), “m”(*__xg(ptr)), “0”(old)
  121. : “memory”);
  122. return prev;
  123. case 2:
  124. asm volatile(LOCK_PREFIX “cmpxchgw %w1,%2”
  125. : “=a”(prev)
  126. : “r”(new), “m”(*__xg(ptr)), “0”(old)
  127. : “memory”);
  128. return prev;
  129. case 4:
  130. asm volatile(LOCK_PREFIX “cmpxchgl %k1,%2”
  131. : “=a”(prev)
  132. : “r”(new), “m”(*__xg(ptr)), “0”(old)
  133. : “memory”);
  134. return prev;
  135. case 8:
  136. asm volatile(LOCK_PREFIX “cmpxchgq %1,%2”
  137. : “=a”(prev)
  138. : “r”(new), “m”(*__xg(ptr)), “0”(old)
  139. : “memory”);
  140. return prev;
  141. }
  142. return old;
  143. }
    atomic_cmpxchg是由cmpxchg指令完成的。它把旧值同atomic_t类型的值相比较，如果相同，就把新值存入atomic_t类型的值中，返回atomic_t类型变量中原有的值。
  144. static inline int atomic_xchg(atomic_t *v, int new)
  145. {
  146. return xchg(&v->counter, new);
  147. }
  149. #define xchg(ptr, v) \
  150. ((typeof((ptr)))__xchg((unsigned long)(v), (ptr), sizeof((ptr))))
  152. static inline unsigned long __xchg(unsigned long x, volatile void *ptr,
  153. int size)
  154. {
  155. switch (size) {
  156. case 1:
  157. asm volatile(“xchgb %b0,%1”
  158. : “=q” (x)
  159. : “m” (*__xg(ptr)), “0” (x)
  160. : “memory”);
  161. break;
  162. case 2:
  163. asm volatile(“xchgw %w0,%1”
  164. : “=r” (x)
  165. : “m” (*__xg(ptr)), “0” (x)
  166. : “memory”);
  167. break;
  168. case 4:
  169. asm volatile(“xchgl %k0,%1”
  170. : “=r” (x)
  171. : “m” (*__xg(ptr)), “0” (x)
  172. : “memory”);
  173. break;
  174. case 8:
  175. asm volatile(“xchgq %0,%1”
  176. : “=r” (x)
  177. : “m” (*__xg(ptr)), “0” (x)
  178. : “memory”);
  179. break;
  180. }
  181. return x;
  182. }
    atomic_xchg则是将新值存入atomic_t类型的变量，并将变量的旧值返回。它使用xchg指令实现。
  183. /**
    • atomic_add_unless - add unless the number is already a given value
    • @v: pointer of type atomic_t
    • @a: the amount to add to v…
    • @u: …unless v is equal to u.
  188. *
    • Atomically adds @a to @v, so long as @v was not already @u.
    • Returns non-zero if @v was not @u, and zero otherwise.
  191. */
  192. static inline int atomic_add_unless(atomic_t *v, int a, int u)
  193. {
  194. int c, old;
  195. c = atomic_read(v);
  196. for (;;) {
  197. if (unlikely(c == (u)))
  198. break;
  199. old = atomic_cmpxchg((v), c, c + (a));
  200. if (likely(old == c))
  201. break;
  202. c = old;
  203. }
  204. return c != (u);
  205. }
    atomic_add_unless的功能比较特殊。它检查v是否等于u，如果不是则把v的值加上a，返回值表示相加前v是否等于u。因为在atomic_read和atomic_cmpxchg中间可能有其它的写操作，所以要循环检查自己的值是否被写进去。
  206. #define atomic_inc_not_zero(v) atomic_add_unless((v), 1, 0)
  208. #define atomic_inc_return(v) (atomic_add_return(1, v))
  209. #define atomic_dec_return(v) (atomic_sub_return(1, v))
    atomic_inc_not_zero在v值不是0时加1。
    atomic_inc_return对v值加1，并返回相加结果。
    atomic_dec_return对v值减1，并返回相减结果。
  210. #define atomic_clear_mask(mask, addr) \
  211. asm volatile(LOCK_PREFIX “andl %0,%1” \
  212. : : “r” (~(mask)), “m” (*(addr)) : “memory”)
    atomic_clear_mask清除变量某些位。
  213. #define atomic_set_mask(mask, addr) \
  214. asm volatile(LOCK_PREFIX “orl %0,%1” \
  215. : : “r” (mask), “m” (*(addr)) : “memory”)
    atomic_set_mask将变量的某些位置位。
  216. / Atomic operations are already serializing on x86 /
  217. #define smp_mb__before_atomic_dec() barrier()
  218. #define smp_mb__after_atomic_dec() barrier()
  219. #define smp_mb__before_atomic_inc() barrier()
  220. #define smp_mb__after_atomic_inc() barrier()
    因为x86的atomic操作大多使用原子指令或者带lock前缀的指令。带lock前缀的指令执行前会完成之前的读写操作，对于原子操作来说不会受之前对同一位置的读写操作，所以这里只是用空操作barrier()代替。barrier()的作用相当于告诉编译器这里有一个内存屏障，放弃在寄存器中的暂存值，重新从内存中读入。
    本节的atomic_t类型操作是最基础的，为了介绍下面的内容，必须先介绍它。如果可以使用atomic_t类型代替临界区操作，也可以加快不少速度。
